int parse_mime_c(const char *cp, Word *ftype, LongWord *atype)
{
  int i;
  int slash;

  if (!cp || !*cp)
    return 0;

  /*
   * two pass
   * 1. type/subtype
   * 2. type
   */

  semi = slash = -1;
  for (i = 0; ; ++i)
  {
    char c = cp[i];
    if (c == 0 || c == ';') break;

    if (c == '/')
    {
      slash = i;
    }
  }

  // try type/subtype
  if (parse_mime(cp, i, ftype, atype)) 
    return 1;

  
  // try type
  if (slash != -1)
    return parse_mime(cp, slash, ftype, atype));

  return 0;
}
